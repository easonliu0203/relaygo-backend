```sql
-- 記錄開始時間

-- 創建測試訂單
INSERT INTO bookings (
  customer_id,
  status,
  pickup_location,
  destination,
  start_date,
  start_time,
  duration_hours,
  vehicle_type,
  total_amount,
  deposit_amount
) VALUES (
  (SELECT id FROM users WHERE role = 'customer' LIMIT 1),
  'pending',
  'Real Test Location A',
  'Real Test Location B',
  CURRENT_DATE + INTERVAL '1 day',
  '14:00:00',
  8,
  'A',
  1000.00,
  300.00
) RETURNING id, created_at;

-- 等待 3 秒後檢查 HTTP 請求
-- （在新的查詢中執行）
SELECT 
  id AS "請求 ID",
  status_code AS "狀態碼",
  created AS "請求時間",
  EXTRACT(EPOCH FROM (created - (SELECT MAX(created_at) FROM bookings WHERE pickup_location LIKE '%Real Test%'))) AS "延遲（秒）"
FROM net._http_response
WHERE created >= NOW() - INTERVAL '10 seconds'
ORDER BY created DESC
LIMIT 1;
```

**預期結果**:
- status_code = 200
- 延遲 < 3 秒
