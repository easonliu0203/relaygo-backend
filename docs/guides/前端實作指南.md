```dart
class BookingFormPage extends StatefulWidget {

class _BookingFormPageState extends State<BookingFormPage> {

              value: duration,
              decoration: InputDecoration(labelText: '使用時數'),
              items: [6, 8, 10, 12].map((hours) => 
                DropdownMenuItem(value: hours, child: Text('${hours}小時'))
              ).toList(),
              onChanged: (value) => setState(() => duration = value!),
            ),
            
            // 上車地點
            TextFormField(
              decoration: InputDecoration(labelText: '上車地點'),
              validator: (value) => value?.isEmpty == true ? '請輸入上車地點' : null,
              onSaved: (value) => pickupLocation = value!,
            ),
            
            Spacer(),
            
            // 提交按鈕
            ElevatedButton(
              onPressed: _submitBooking,
              child: Text('確認預約'),
              style: ElevatedButton.styleFrom(
                minimumSize: Size(double.infinity, 50),
              ),
            ),
          ],
        ),
      ),
    );
  }
  
  Future<void> _submitBooking() async {
    if (_formKey.currentState!.validate()) {
      _formKey.currentState!.save();
      
      try {
        final response = await BookingService.createBooking({
          'vehicleType': vehicleType,
          'startDate': DateFormat('yyyy-MM-dd').format(selectedDate),
          'startTime': selectedTime.format(context),
          'duration': duration,
          'pickupLocation': pickupLocation,
        });
        
        if (response.success) {
          Navigator.pushReplacement(
            context,
            MaterialPageRoute(
              builder: (context) => PaymentPage(
                booking: response.data.booking,
                pricing: response.data.pricing,
              ),
            ),
          );
        }
      } catch (error) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(content: Text('預約失敗：$error')),
        );
      }
    }
  }
}
```

```dart
              onPressed: isProcessing ? null : _processPayment,
              child: isProcessing 
                ? CircularProgressIndicator(color: Colors.white)
                : Text('支付訂金 NT\$ ${widget.pricing.depositAmount}'),
              style: ElevatedButton.styleFrom(
                minimumSize: Size(double.infinity, 50),
                backgroundColor: Colors.green,
              ),
            ),
          ),
        ],
      ),
    );
  }
  
  Future<void> _processPayment() async {
    setState(() => isProcessing = true);
    
    try {
      final response = await PaymentService.payDeposit(
        widget.booking.id,
        paymentMethod: 'mock', // 封測階段使用模擬支付
      );
      
      if (response.success) {
        if (response.data.autoCompleted) {
          // 模擬支付自動完成
          _showPaymentSuccess();
        } else {
          // 顯示支付說明
          _showPaymentInstructions(response.data);
        }

        SnackBar(content: Text('支付失敗：$error')),
      );

        ],
      ),
    );
  }
```

```dart
  void initState() {
    super.initState();
    _loadBookingStatus();
    _subscribeToStatusUpdates();
  }
  
  @override
  Widget build(BuildContext context) {
    if (bookingStatus == null) {
      return Scaffold(
        body: Center(child: CircularProgressIndicator()),
      );
    }
    
    return Scaffold(
      appBar: AppBar(title: Text('行程狀態')),
      body: Column(
        children: [
          // 狀態進度條
          _buildStatusProgress(),
          
          // 當前狀態資訊
          _buildCurrentStatusInfo(),
          
          // 司機資訊 (如果已派單)
          if (bookingStatus!.driverId != null) _buildDriverInfo(),
          
          // 聊天按鈕 (如果聊天室已開啟)
          if (bookingStatus!.chatRoomId != null) _buildChatButton(),
          
          // 行動按鈕
          _buildActionButtons(),
        ],
      ),
    );
  }
  
  Widget _buildActionButtons() {
    switch (bookingStatus!.status) {
      case 'driver_arrived':
        return ElevatedButton(
          onPressed: _startTrip,
          child: Text('開始行程'),
          style: ElevatedButton.styleFrom(backgroundColor: Colors.green),
        );
        
      case 'trip_started':
        return ElevatedButton(
          onPressed: _endTrip,
          child: Text('結束行程'),
          style: ElevatedButton.styleFrom(backgroundColor: Colors.red),
        );
        
      default:
        return SizedBox.shrink();
    }
  }
  
  Future<void> _startTrip() async {
    try {
      await BookingService.startTrip(widget.bookingId);
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('行程已開始')),
      );
    } catch (error) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('操作失敗：$error')),
      );
    }
  }
  
  Future<void> _endTrip() async {
    try {
      final response = await BookingService.endTrip(widget.bookingId);
      
      Navigator.pushReplacement(
        context,
        MaterialPageRoute(
          builder: (context) => PaymentBalancePage(
            bookingId: widget.bookingId,
            finalAmount: response.data.finalAmount,
          ),
        ),
      );
    } catch (error) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('操作失敗：$error')),
      );
    }
  }
}
```

```tsx
// pages/admin/dispatch.tsx
import { useState, useEffect } from 'react';
import { Card, Button, Select, Table, Modal, message } from 'antd';

interface PendingBooking {
  id: string;
  bookingNumber: string;
  customerName: string;
  vehicleType: string;
  startDate: string;
  startTime: string;
  pickupLocation: string;
  createdAt: string;
}

interface AvailableDriver {
  id: string;
  name: string;
  vehicleType: string;
  rating: number;
  distance: number;
  isAvailable: boolean;
}

export default function DispatchPage() {
  const [pendingBookings, setPendingBookings] = useState<PendingBooking[]>([]);
  const [availableDrivers, setAvailableDrivers] = useState<AvailableDriver[]>([]);
  const [selectedBooking, setSelectedBooking] = useState<PendingBooking | null>(null);
  const [isModalVisible, setIsModalVisible] = useState(false);
  const [selectedDriver, setSelectedDriver] = useState<string>('');
  
  useEffect(() => {
    loadPendingBookings();
  }, []);
  
  const loadPendingBookings = async () => {
    try {
      const response = await fetch('/api/booking-flow/admin/bookings/pending-dispatch');
      const data = await response.json();
      setPendingBookings(data.data.bookings);
    } catch (error) {
      message.error('載入待派單列表失敗');
    }
  };
  
  const handleManualDispatch = (booking: PendingBooking) => {
    setSelectedBooking(booking);
    loadAvailableDrivers(booking.vehicleType);
    setIsModalVisible(true);
  };
  
  const handleAutoDispatch = async (bookingId: string) => {
    try {
      const response = await fetch(`/api/booking-flow/admin/bookings/${bookingId}/auto-dispatch`, {
        method: 'POST',
      });
      
      const result = await response.json();
      
      if (result.success) {
        message.success('自動派單成功');
        loadPendingBookings();
      } else {
        message.error(`自動派單失敗：${result.message}`);
      }
    } catch (error) {
      message.error('自動派單失敗');
    }
  };
  
  const confirmManualDispatch = async () => {
    if (!selectedBooking || !selectedDriver) return;
    
    try {
      const response = await fetch(`/api/booking-flow/admin/bookings/${selectedBooking.id}/assign`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ driverId: selectedDriver }),
      });
      
      const result = await response.json();
      
      if (result.success) {
        message.success('手動派單成功');
        setIsModalVisible(false);
        loadPendingBookings();
      } else {
        message.error(`手動派單失敗：${result.message}`);
      }
    } catch (error) {
      message.error('手動派單失敗');
    }
  };
  
  const columns = [
    {
      title: '訂單編號',
      dataIndex: 'bookingNumber',
      key: 'bookingNumber',
    },
    {
      title: '客戶姓名',
      dataIndex: 'customerName',
      key: 'customerName',
    },
    {
      title: '車型',
      dataIndex: 'vehicleType',
      key: 'vehicleType',
    },
    {
      title: '出發時間',
      key: 'startDateTime',
      render: (record: PendingBooking) => `${record.startDate} ${record.startTime}`,
    },
    {
      title: '上車地點',
      dataIndex: 'pickupLocation',
      key: 'pickupLocation',
    },
    {
      title: '操作',
      key: 'actions',
      render: (record: PendingBooking) => (
        <div>
          <Button 
            type="primary" 
            size="small" 
            onClick={() => handleManualDispatch(record)}
            style={{ marginRight: 8 }}
          >
            手動派單
          </Button>
          <Button 
            type="default" 
            size="small" 
            onClick={() => handleAutoDispatch(record.id)}
          >
            自動派單
          </Button>
        </div>
      ),
    },
  ];
  
  return (
    <div>
      <Card title="派單管理" style={{ margin: 16 }}>
        <Table 
          columns={columns} 
          dataSource={pendingBookings}
          rowKey="id"
          pagination={{ pageSize: 10 }}
        />
      </Card>
      
      <Modal
        title="手動派單"
        visible={isModalVisible}
        onOk={confirmManualDispatch}
        onCancel={() => setIsModalVisible(false)}
        okText="確認派單"
        cancelText="取消"
      >
        {selectedBooking && (
          <div>
            <p><strong>訂單：</strong>{selectedBooking.bookingNumber}</p>
            <p><strong>車型：</strong>{selectedBooking.vehicleType}</p>
            <p><strong>時間：</strong>{selectedBooking.startDate} {selectedBooking.startTime}</p>
            
            <div style={{ marginTop: 16 }}>
              <label>選擇司機：</label>
              <Select
                style={{ width: '100%', marginTop: 8 }}
                placeholder="請選擇司機"
                value={selectedDriver}
                onChange={setSelectedDriver}
              >
                {availableDrivers.map(driver => (
                  <Select.Option key={driver.id} value={driver.id}>
                    {driver.name} - 評分: {driver.rating} - 距離: {driver.distance}km
                  </Select.Option>
                ))}
              </Select>
            </div>
          </div>
        )}
      </Modal>
    </div>
  );
}
```

```dart
// lib/services/websocket_service.dart
class WebSocketService {
  static final WebSocketService _instance = WebSocketService._internal();
  factory WebSocketService() => _instance;
  WebSocketService._internal();
  
  IOWebSocketChannel? _channel;
  StreamController<Map<String, dynamic>> _messageController = StreamController.broadcast();
  
  Stream<Map<String, dynamic>> get messageStream => _messageController.stream;
  
  void connect(String userId, String userType) {
    _channel = IOWebSocketChannel.connect('ws://localhost:3000');
    
    // 加入用戶房間
    _channel!.sink.add(jsonEncode({
      'event': 'join',
      'data': {
        'userType': userType,
        'userId': userId,
      }
    }));
    
    // 監聽訊息
    _channel!.stream.listen((message) {
      final data = jsonDecode(message);
      _messageController.add(data);
    });
  }
  
  void joinChatRoom(String chatRoomId, String userId) {
    _channel?.sink.add(jsonEncode({
      'event': 'join_chat',
      'data': {
        'chatRoomId': chatRoomId,
        'userId': userId,
      }
    }));
  }
  
  void sendMessage(String chatRoomId, String senderId, String senderType, String content) {
    _channel?.sink.add(jsonEncode({
      'event': 'send_message',
      'data': {
        'chatRoomId': chatRoomId,
        'senderId': senderId,
        'senderType': senderType,
        'type': 'text',
        'content': content,
      }
    }));
  }
  
  void disconnect() {
    _channel?.sink.close();
    _messageController.close();
  }
}
```

### API 測試腳本

```bash
#!/bin/bash
# test_booking_flow.sh

BASE_URL="http://localhost:3000/api/booking-flow"
CUSTOMER_TOKEN="your-customer-jwt-token"
DRIVER_TOKEN="your-driver-jwt-token"
ADMIN_TOKEN="your-admin-jwt-token"

echo "=== 測試完整預約流程 ==="

# 1. 創建預約
echo "1. 創建預約..."
BOOKING_RESPONSE=$(curl -s -X POST "$BASE_URL/bookings" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "vehicleType": "A",
    "startDate": "2025-02-01",
    "startTime": "09:00",
    "duration": 8,
    "pickupLocation": "台北車站"
  }')

BOOKING_ID=$(echo $BOOKING_RESPONSE | jq -r '.data.booking.id')
echo "預約ID: $BOOKING_ID"

# 2. 支付訂金
echo "2. 支付訂金..."
curl -s -X POST "$BASE_URL/bookings/$BOOKING_ID/deposit" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"paymentMethod": "mock"}' | jq

# 3. 手動派單
echo "3. 手動派單..."
curl -s -X POST "$BASE_URL/admin/bookings/$BOOKING_ID/assign" \
  -H "Authorization: Bearer $ADMIN_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"driverId": "driver-uuid-here"}' | jq

# 4. 司機接單
echo "4. 司機接單..."
curl -s -X POST "$BASE_URL/bookings/$BOOKING_ID/accept" \
  -H "Authorization: Bearer $DRIVER_TOKEN" | jq

# 5. 司機出發
echo "5. 司機出發..."
curl -s -X POST "$BASE_URL/bookings/$BOOKING_ID/depart" \
  -H "Authorization: Bearer $DRIVER_TOKEN" | jq

# 6. 司機到達
echo "6. 司機到達..."
curl -s -X POST "$BASE_URL/bookings/$BOOKING_ID/arrive" \
  -H "Authorization: Bearer $DRIVER_TOKEN" | jq

# 7. 客戶開始行程
echo "7. 客戶開始行程..."
curl -s -X POST "$BASE_URL/bookings/$BOOKING_ID/start-trip" \
  -H "Authorization: Bearer $CUSTOMER_TOKEN" | jq
```
